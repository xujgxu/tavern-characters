      document.addEventListener('DOMContentLoaded', () => {

        // 核心修改 3：从 script 标签获取文本内容，而不是 dataset
        const rawDataElement = document.getElementById('raw-options-data');

        if (optionsContainer && rawDataElement) {
          const optionsString = rawDataElement.textContent; // 获取纯文本，不受引号影响
          const optionMatches = optionsString.matchAll(/<(option(?:_\d+)?)>([\s\S]*?)<\/\1>/g);
          let hasOptions = false;

          for (const match of optionMatches) {
            const optionText = match[2].trim();
            if (optionText) {
              hasOptions = true;
              const optionDiv = document.createElement('div');
              optionDiv.className = 'action-option';
              optionDiv.textContent = optionText;
              optionDiv.onclick = event => {
                const text = event.currentTarget.textContent.trim();
                if (typeof triggerSlash === 'function') {
                  triggerSlash(`/setinput ${text}`);
                }
              };
              optionsContainer.appendChild(optionDiv);
            }
          }
          if (!hasOptions) {
            const emptyState = document.createElement('div');
            emptyState.className = 'empty-state';
            emptyState.textContent = '无可用选项';
            optionsContainer.innerHTML = '';
            optionsContainer.appendChild(emptyState);
          }
        }
      });
